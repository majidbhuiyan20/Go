12. postgres=# UPDATE cars
SET color = CASE
    WHEN brand = 'Ford' AND model = 'Mustang' THEN 'Red'
    WHEN brand = 'Volvo' AND model = 'p1800' THEN 'Blue'
    WHEN brand = 'BMW' AND model = 'M1' THEN 'Black'
    WHEN brand = 'Toyota' AND model = 'Celica' THEN 'Green'
END;
UPDATE 4
